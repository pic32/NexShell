109 SHELL_RESULT AddUserCommand(const char* CommandName, const char* Description, const char* HelpFile, SHELL_RESULT(*ExecuteFile)(char* [], UINT32, PIPE*))
110 {
111 UINT32 i;
112 COMMAND_INFO* CommandInfo;
113
114 // validate the minimum data required
115 if (CommandName == NULL || ExecuteFile == NULL)
116 return SHELL_INVALID_ARGUMENT;
117
118 // does the command already exist?
119 if (LinkedListContains(&gUserCommandList, CommandName, 0, 0) != 0)
120 {
121 // it already exists
122 return SHELL_COMMAND_ALREADY_EXISTS;
123 }
124
125 // is it in the regular command list?
126 for (i = 0; i < sizeof(gCommandList) / sizeof(COMMAND_INFO); i++)
127 {
128 if (strcmp(CommandName, gCommandList[i].CommandName) == 0)
129 return SHELL_COMMAND_ALREADY_EXISTS;
130 }
131
132 // get the space for the structure
133 CommandInfo = NexShellMalloc(sizeof(COMMAND_INFO));
134
135 // did we get the space?
136 if (CommandInfo == NULL)
137 return SHELL_MALLOC_FAILURE;
138
139 // add it to the list
140 if (LinkedListAdd(&gUserCommandList, CommandInfo) == FALSE)
141 {
142 // we didn't get the space, free this space
143 NexShellFreeMethod(CommandInfo);
144
145 return SHELL_LINKED_LIST_OPERATION_FAILURE;
146 }
147
148 // initialize the data since we know we have all the space
149 CommandInfo->CommandName = (char*)CommandName;
150 CommandInfo->Description = (char*)Description;
151 CommandInfo->Help = (char*)HelpFile;
152 CommandInfo->ExecuteFile = ExecuteFile;
153
154 return SHELL_SUCCESS;
155 }
